syntax

mesh syntax is minimal. pipelines flow left to right. indentation defines blocks.

values

typesyntaxexample
string"text" or 'text'"hello"
numberdigits42, 3.14
booleantrue, falsetrue
nullnullnull
field.field.name, .items
index[n][0], [-1]
slice[start:end][:5], [1:3]

tool calls

tool_name arg1 arg2 key=value --flag

tool names can contain dots (http.get, json.parse). positional args come first, then key=value pairs, then --flags.

pipes

step_a → step_b → step_c

data flows left to right. output of one step is input to the next.

conditionals

if .status == 200:
    log "ok"
→ else:
    log "failed"

loops

# for each item
json.parse → for each items:
    format "{{.name}}"

# timed loop
loop every 60s:
    http.get "https://example.com/health"

parallel

parallel:
  branch users:
    http.get "url/users" → json.parse
  branch posts:
    http.get "url/posts" → json.parse
→ merge
→ print

error handling

retry 3, backoff 2:
    http.get "https://flaky.example.com"
→ on_error:
    log "failed"
    return {ok: false}

informal grammar

program     ::= statement*
statement   ::= pipeline | import | tool_def | parallel | conditional | loop | retry_block
pipeline    ::= step ( "→" step )*
step        ::= tool_call | ref | value
tool_call   ::= NAME (arg | KEYWORD "=" value | "--" FLAG)*
ref         ::= "." FIELD ("[" (INT | slice) "]")?
value       ::= STRING | NUMBER | "true" | "false" | "null"
import      ::= "import" STRING
tool_def    ::= "tool" NAME ":" block
parallel    ::= "parallel:" ( "branch" NAME ":" block )+
conditional ::= "if" expr ":" block ( "→" "else:" block )?
loop        ::= "loop" "every" INT "s:" block
retry_block ::= "retry" INT ("," "backoff" INT)? ":" block ( "→" "on_error:" block )?
block       ::= INDENT statement+ DEDENT
